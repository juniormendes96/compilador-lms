PROGRAM maior_numero;

    VAR numero1, numero2: INTEGER;

    BEGIN
        READLN(numero1; (* aqui deve ocorrer o erro *)
        READLN(numero2);
        IF numero1 > numero2 THEN
            WRITELN("Primeiro número é maior")
        ELSE IF numero2 > numero1 THEN
            WRITELN("Segundo número é maior")
        ELSE
            WRITELN("Os números são iguais");
END.
